## Getting Started

### Installation

```jsx
npm install @brightcodeui/beta-ui
# or
yarn add @brightcodeui/beta-ui
```


### Usage

```jsx
import { Button } from '@brightcodeui/beta-ui';

function App() {
  return (
    <div>
      <Button variant="red">Click Me</Button>
    </div>
  );
}

export default App;
```
